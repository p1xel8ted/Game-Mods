[01. Trait Replacement]

## Replace negative traits with positive ones on all followers (existing and new).
# Setting type: Boolean
# Default value: false
Enable Trait Replacement = true

## Only use traits you have unlocked. Applies to both trait replacement and new follower trait selection.
# Setting type: Boolean
# Default value: true
Use Unlocked Traits Only = false

## Merge all trait pools into one, bypassing vanilla's normal/rare split. Without this, vanilla assigns traits from separate pools (normal pool for first 2 traits, ~20% chance of rare pool for 3rd trait). Enable this for trait weights to have full control over distribution. Unique traits require their individual toggles.
# Setting type: Boolean
# Default value: false
Use All Traits Pool = true

## When replacing negative traits, exclusive traits (like Lazy) are replaced with their positive counterpart (Industrious) instead of a random trait.
# Setting type: Boolean
# Default value: true
Prefer Exclusive Counterparts = true

## Minimum number of traits new followers will have. Vanilla is 2.
# Setting type: Int32
# Default value: 2
# Acceptable value range: From 2 to 8
Minimum Traits = 2

## Maximum number of traits new followers will have. Vanilla is 3. Limited to 8 due to UI constraints.
# Setting type: Int32
# Default value: 3
# Acceptable value range: From 2 to 8
Maximum Traits = 3

## When re-indoctrinating an existing follower (at the altar), randomize their traits using the configured min/max. Vanilla re-indoctrination only changes appearance/name.
# Setting type: Boolean
# Default value: false
Randomize Traits on Re-indoctrination = false

## Adds the Re-educate command to normal followers. Using it will re-roll their traits using the configured min/max and weights.
# Setting type: Boolean
# Default value: false
Trait Reroll via Reeducation = false

[04. Trait Weights]

## Enable weighted random selection for new followers. Weights affect trait selection within each pool. For full control over all traits, enable 'Use All Traits Pool' - otherwise vanilla's normal/rare pool split still applies (rare pool has ~20% chance for 3rd trait). Set a weight to 0 to disable a trait.
# Setting type: Boolean
# Default value: false
Enable Trait Weights = true

## Include traits normally granted through gameplay events (marriage, parenting, criminal, missionary, etc.) in the weights list. Only applies when 'Use All Traits Pool' is enabled. Warning: This can result in nonsensical assignments (e.g., ProudParent on followers who have never had children).
# Setting type: Boolean
# Default value: false
Include Event Traits = true
